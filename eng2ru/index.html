An English to Russian Transliterator

The Program

function transliterate(phrase)
{
    phrase = phrase.toLowerCase();

    let ENG_TO_RU_LETTERS = {'a':'а', 'b':'б', 'v':'в', 'g':'г', 'd':'д',
        'e':'е', 'z':'з', 'i':'и', 'k':'к', 'l':'л', 'm':'м', 'n':'н',
        'o':'о', 'p':'п', 'r':'р', 's':'с', 't':'т', 'f':'ф', 'h':'х',
        'u':'у', 'j':'й', 'y':'й', 'w':'в', 'c':'к'};

    let ENG_TO_RU_DIPHTHONGS = {"ch":"ч", "yu":"ю", "ts":"ц", "ya":"я",
        "sh":"ш", "oo":"у", "yo":"ё", "ih":"ы", "ly ":"ли ", "ly":"лЬ",
        "zh":"ж", "eh":"э", "ee":"и", "wh":"в", "i ":"aй "};

    for(var key in ENG_TO_RU_DIPHTHONGS)
        phrase = phrase.replace(key, ENG_TO_RU_DIPHTHONGS[key])

    let replacement;
    for(let i = 0; i < phrase.length; ++i)
    {
        replacement = ENG_TO_RU_LETTERS[phrase[i]];
        if(replacement != null)
            phrase = phrase.substring(0, i) + replacement + phrase.substring(i + 1);
    }

    return phrase;
}
